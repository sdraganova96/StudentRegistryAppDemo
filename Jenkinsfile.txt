pipeline {
    agent any

    environment {
        NODE_VERSION = 'NodeJS21.7.2'
        SERVICE_ID = 'render-service-id'
        RENDER_API_KEY = 'render-api-key'
    }

    tools {
        nodejs "${NODE_VERSION}"
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Install dependencies') {
            steps {
                sh 'npm install'
                sh 'npm install -g wait-on kill-port'
            }
        }

        stage('Run tests') {
            steps {
                sh 'start /b npm run start'
                sh 'wait-on http://localhost:8081'
                sh 'npm test'
                sh 'kill-port 8081'
            }
        }
    }

    post {
        always {
            echo 'CI Pipeline completed.'
        }
        failure {
            sh 'kill-port 8081'
        }
    }
}
